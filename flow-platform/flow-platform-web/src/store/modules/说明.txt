4. 模块化 Vuex Store（可选）
如果项目较大，可以将 Vuex Store 拆分为多个模块。例如：

(1) 用户模块：store/modules/user.js

(2) 认证模块：store/modules/auth.js

(3) 在 store/index.js 中引入模块
import { createStore } from 'vuex';
import userModule from './modules/user';
import authModule from './modules/auth';

const store = createStore({
  modules: {
    user: userModule,
    auth: authModule,
  },
});

export default store;

(4) 在组件中使用模块
export default {
  computed: {
    user() {
      return this.$store.state.user.user;
    },
    isAuthenticated() {
      return this.$store.state.auth.isAuthenticated;
    },
  },
  methods: {
    login(credentials) {
      this.$store.dispatch('auth/login', credentials.token)
        .then(() => {
          this.$store.dispatch('user/setUser', credentials.user);
          this.$router.push('/dashboard');
        });
    },
  },
};
