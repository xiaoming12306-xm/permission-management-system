在 App.vue 中使用 <router-view> 渲染路由组件：
<template>
  <div id="app">
    <router-view></router-view>
  </div>
</template>

<script>
export default {
  name: 'App',
};
</script>

<style>
/* 全局样式 */
</style>


在组件中使用 this.$router.push 进行路由跳转：
this.$router.push({ name: 'Dashboard' });
或使用 <router-link> 组件：
<router-link to="/dashboard">Dashboard</router-link>
